\subsection{Изменения в интерфейсных спецификациях}

\begin{description}
\item [1.6.3]: нет.
\item [1.6.2]: нет.
\item [1.6.1]: 
      \begin{component}{ptrs}
      \item изменен шаблон \verb|owned_ptr| - добавлен метод \verb|get_ownity|
      \end{component}
\item [1.6.0]: 
      \begin{component}{Threading}
      \item Изменен интерфейс  \verb|Thread| - добавлен метод \verb|getThreadContext|
      \item добавлен интерфейс \verb|ThreadContext|
      \end{component}
\item [1.5.0]: 
      \begin{component}{Threading}
      \item добавлен шаблон \verb|counted_mt_ptr|;
      \item изменены спецификации исключений;
      \end{component}
      \begin{component}{DynamicModules}
      \item расширена спецификация;
      \end{component}
      \begin{component}{ProgOptions}
      \item методы ProgOptions::parseFile и ProgOptions::parseString теперь имеют два параметра,
            второй - это имя выполняемой программы, по умолчанию "unspecified".
      \item изменился интерфейс и порядок использования класса ProgOptions::ArgsHolder: 
            \begin{itemize}
            \item конструктор ArgsHolder(int,char**) и вызывавший его метод
                  ProgOptions::createArgsHolder() удалены;
            \item вместо них предоставлены:
                  \begin{enumerate}
                  \item конструктор ArgsHolder() (позабытый в прошлых версиях);
                  \item метод ArgsHolder::takeArgv(const ProgOptions\&);
                  \end{enumerate}
            \end{itemize}
      \end{component}

\item [1.4.1]:

      \begin{component}{ProgOptions}
      \item добавлен класс ProgOptions::ArgsHolder и метод ProgOptions::createArgsHolder(), 
            для получения экземпляра этого класса
      \end{component}

\item [1.4.0]: 

      \begin{component}{Общие требования}
      \item использование пространств имен C++ стало обязательным.
      \item пространство имен C++ не импортируется во включаемых файлах пакета.
      \end{component}

      \begin{component}{ServiceOptions}
      \item включаемый файл ServiceOptions заменен на ServiceOptions.h
      \end{component}

\item [1.2.0]:

      \begin{component}{Threading}
      \item Общедоступные статические методы Thread::sleep() и Thread::nanosleep()
            перестали быть статическими и стали защищенными; вместо них появились
            общедоступные статические методы Thread::sleepCurrent() и Thread::nanosleepCurrent().
      \end{component}
      \begin{component}{Logger}
      \item Включаемый файл \verb|Logger| удален.
      \item Файл \verb|Logger.h| не импортирует определения в текущее пространство имен.
      \item Добавлен методы \verb|Logger::setDuppedToStderr(bool)| и 
            \verb|Logger::setSyslogOutput(bool);|
      \item Добавлены блокировщики вывода в потоки Logger.
      \end{component}

\item [1.1.0]:

      \begin{component}{Threading}
      \item добавлены спецификации исключений во все методы.
      \end{component}

\item [1.0.3]:

      \begin{component}{ProgOptions}
      \item добавлен метод \verb|ProgOptions::allowUnknownOptions|
      \end{component}

\end{description}
